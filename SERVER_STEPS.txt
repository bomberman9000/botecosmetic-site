# Команды для выполнения на сервере (выполняйте по порядку):

# ШАГ 1: Проверяем, есть ли архив
ls -lh /tmp/deploy.tar.gz

# Если архива НЕТ - нужно сначала скопировать его с вашего компьютера
# (выйдите с сервера: exit, затем на вашем компьютере выполните: scp deploy.tar.gz root@144.31.64.130:/tmp/)

# ШАГ 2: Переходим в правильную директорию
mkdir -p /var/www/bote-site
cd /var/www/bote-site

# ШАГ 3: Распаковываем архив
tar -xzf /tmp/deploy.tar.gz

# ШАГ 4: Устанавливаем зависимости
npm install

# ШАГ 5: Собираем проект
npm run build

# ШАГ 6: Останавливаем старое приложение (если есть)
pm2 delete bote-site 2>/dev/null || true

# ШАГ 7: Запускаем приложение
pm2 start npm --name "bote-site" -- start
pm2 save

# ШАГ 8: Проверяем статус
pm2 list
