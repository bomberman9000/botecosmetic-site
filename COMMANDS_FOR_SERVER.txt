# Команды для выполнения на сервере
# Скопируйте и выполните эти команды в веб-консоли или через доступ к серверу

# 1. Перейдите в директорию проекта
mkdir -p /var/www/bote-site
cd /var/www/bote-site

# 2. Если архив уже загружен в /tmp/, распакуйте его:
tar -xzf /tmp/deploy.tar.gz

# 3. Установите зависимости (важно: без --production!)
npm install

# 4. Соберите проект
npm run build

# 5. Установите PM2, если не установлен
npm install -g pm2

# 6. Остановите старое приложение (если есть)
pm2 delete bote-site 2>/dev/null || true

# 7. Запустите приложение
pm2 start npm --name "bote-site" -- start

# 8. Сохраните конфигурацию PM2
pm2 save

# 9. Проверьте статус
pm2 list

# 10. Посмотрите логи (опционально)
pm2 logs bote-site --lines 20

# После выполнения всех команд сайт будет доступен по адресу:
# http://144.31.64.130:3001
